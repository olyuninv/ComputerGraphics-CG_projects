#version 330

uniform vec4 Ambient;

in vec3 n_eye;
in vec2 TexCoord0;

out vec4 frag_colour;

struct DirectionalLight
{
   	vec3 Color;
   	float AmbientIntensity;
};

uniform DirectionalLight gDirectionalLight;

void main(){
	frag_colour = min(vec4 (n_eye, 1.0) * vec4(gDirectionalLight.Color, 1.0f) * gDirectionalLight.AmbientIntensity, vec4(1.0));
}
